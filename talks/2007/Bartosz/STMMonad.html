The STM Monad

    • Monad (support "do" notation and sequencing) 
 data STM a
 instance Monad STM
    • Exceptions 
 throw :: Exception -> STM a
 catch :: STM a -> (Exception->STM a) -> STM a
    • STM computations 
 atomic :: STM a -> IO a
 retry :: STM a
 orElse :: STM a -> STM a -> STM a
    • Transactional variables 
 data TVar a
 newTVar :: a -> STM (TVar a)
 readTVar :: TVar a -> STM a
 writeTVar :: TVar a -> a -> STM ()
    • It's an instance of MonadPlus (whatever that means)
